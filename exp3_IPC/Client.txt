import java.io.*;
import java.net.*;
public class Client {
public static void main(String[] args) {
try{
Socket s=new Socket("0.0.0.0",6666);
DataOutputStream dout=new DataOutputStream(s.getOutputStream());
dout.writeUTF("Hello My Boyy");
dout.flush();
dout.close();
s.close();
}
catch(Exception e){System.out.println(e);}
}
}
